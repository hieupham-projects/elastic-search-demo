# Sum metric
# Fetching the total number of critical patients
GET covid/_search
{
  "size": 0,
  "aggs": {
    "critical_patients": {
      "sum": {
        "field": "critical"
      }
    }
  }
}

# Max metric
# The query to fetch the highest number of deaths among the 10 countries we have in our data set:
GET covid/_search
{
  "size": 0, 
  "aggs": {
    "total_deaths": {
      "max": {
        "field": "deaths"
      }
    }
  }
}

# Stats metric
# We can find the minimum (min), average (avg), and others too. But thereâ€™s one statistical function that returns all these basic metrics in one go: the stats metric:
GET covid/_search
{
  "size": 0, 
  "aggs": {
    "all_stats": {
      "stats": {
        "field": "deaths"
      }
    }
  }
}


# Bucketing aggregations

# Histogram buckets
#Fetching the countries by number of critical patients in buckets of 2500:
GET covid/_search
{
  "size": 0,
  "aggs": {
    "critical_patients_as_histogram": {
      "histogram": {
        "field": "critical",
        "interval": 3000
      }
    }
  }
}

# Range buckets
# Casualties by custom ranges using range bucketing:
GET covid/_search
{
  "size": 0, 
  "aggs": {
    "range_countries": {
      "range": {
        "field": "deaths",
        "ranges": [
          {"to": 60000},
          {"from": 60000,"to": 70000},
          {"from": 70000,"to": 80000},
          {"from": 80000,"to": 120000}
        ]
      }
    }
  }
}
